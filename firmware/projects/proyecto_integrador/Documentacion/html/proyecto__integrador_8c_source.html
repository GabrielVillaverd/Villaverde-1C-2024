107void Manejo_Datos(uint8_t *data, uint8_t length)
108{
109 uint8_t i = 1;
110 static float temperat = 0, luz = 0, humedad = 0;
111
112 if (data[0] == 't')
113 {
115 temperat = 0;
116 while (data[i] != 'A')
117 {
119 temperat = temperat * 10;
120 temperat = temperat + (data[i] - '0'); // Resto el valor ASCII del 0 para obtener el numero
121 i++;
122 }
123 }
124 else if (data[0] == 'h')
125 {
127 humedad = 0;
128 while (data[i] != 'A')
129 {
131 humedad = humedad * 10;
132 humedad = humedad + (data[i] - '0');
133 i++;
134 }
135 }
136 else if (data[0] == 'l')
137 {
139 luz = 0;
140 while (data[i] != 'A')
141 {
143 luz = luz * 10;
144 luz = luz + (data[i] - '0');
145 i++;
146 }
147 }
149 parametro_temperatura = temperat;
150 parametro_humedad = humedad;
151 parametro_luz = luz;
152 umbral_temperatura_bajo = parametro_temperatura - 2.0;
153 umbral_temperatura_alto = parametro_temperatura + 2.0;
154 umbral_humedad_alto = parametro_humedad;
155
156 char cadena_hum[128];
157 strcpy(cadena_hum, "");
158 sprintf(cadena_hum, "*r%.2f\n*", parametro_humedad); // muestro el parametro humedad por ble utilizando la letra 'r'
159 BleSendString(cadena_hum);
160
161 char cadena_temp[128];
162 strcpy(cadena_temp, "");
163 sprintf(cadena_temp, "*c%.2f\n*", parametro_temperatura); // muestro el parametro temperatura por ble utilizando la letra 'c'

240void Calefaccion(void *pvParameter)
241{
242 float temperatura_tarea = 0;
243 int i = 0;
244 while (1)
245 {
246 ulTaskNotifyTake(pdTRUE, portMAX_DELAY);
247 temperatura_tarea += Si7007MeasureTemperature();
248 i++;
249 if (i == 3)
250 {
251 temperatura_tarea = (temperatura_tarea / 3);
252 char cadena_temp[128];
253 strcpy(cadena_temp, "");
254 sprintf(cadena_temp, "*T%.2f\n*", temperatura_tarea);
255 BleSendString(cadena_temp);
256 LedOn(LED_2);
257 if (temperatura_tarea < umbral_temperatura_bajo)
258 {
259 LedOff(LED_2);
260 LedOn(LED_3);
261 printf("\nSe prende aire en calor\n");
264 }
265 else
266 LedOff(LED_3);
267 if (temperatura_tarea > umbral_temperatura_alto)
268 {
269 LedOff(LED_2);
270 LedOn(LED_1);
271 printf("\nSe prende aire en frio");
274 }
275 else
276 LedOff(LED_1);
277
278 i = 0;
279 temperatura_tarea = 0;
280 }
281 }
